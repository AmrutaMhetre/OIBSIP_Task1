    <script>
        $(document).ready(function(){
            function isPrime(num) {
                if (num <= 1) return false;
                if (num <= 3) return true;
                if (num % 2 === 0 || num % 3 === 0) return false;
                for (let i = 5; i * i <= num; i += 6) {
                    if (num % i === 0 || num % (i + 2) === 0) return false;
                }
                return true;
            }

            function isArmstrong(num) {
                let sum = 0;
                let temp = num;
                const n = num.toString().length;
                while (temp > 0) {
                    let digit = temp % 10;
                    sum += Math.pow(digit, n);
                    temp = Math.floor(temp / 10);
                }
                return sum === num;
            }

            function factorial(num) {
                if (num === 0) return 1;
                let fact = 1;
                for (let i = 1; i <= num; i++) {
                    fact *= i;
                }
                return fact;
            }

            $('#checkPrime').click(function() {
                let num = parseInt($('#inputNumber').val());
                if (isNaN(num)) {
                    $('#resultArea').text('Please enter a valid number.');
                } else {
                    let result = isPrime(num) ? `${num} is a prime number.` : `${num} is not a prime number.`;
                    $('#resultArea').text(result);
                }
            });

            $('#checkArmstrong').click(function() {
                let num = parseInt($('#inputNumber').val());
                if (isNaN(num)) {
                    $('#resultArea').text('Please enter a valid number.');
                } else {
                    let result = isArmstrong(num) ? `${num} is an Armstrong number.` : `${num} is not an Armstrong number.`;
                    $('#resultArea').text(result);
                }
            });

            $('#findFactorial').click(function() {
                let num = parseInt($('#inputNumber').val());
                if (isNaN(num)) {
                    $('#resultArea').text('Please enter a valid number.');
                } else {
                    let result = `Factorial of ${num} is ${factorial(num)}.`;
                    $('#resultArea').text(result);
                }
            });
        });
    </script>
</body>

    <script>
        function validatePassword() {
            var lowercaseChecked = document.getElementById("lowercase").checked;
            var uppercaseChecked = document.getElementById("uppercase").checked;
            var digitChecked = document.getElementById("digit").checked;
            var specialCharChecked = document.getElementById("specialChar").checked;
            var minLengthChecked = document.getElementById("minLength").checked;

            if (lowercaseChecked && uppercaseChecked && digitChecked && specialCharChecked && minLengthChecked) {
                return true;
            } else {
                alert("Please enter the password with all the given validations");
                return false;
            }
        }

        function checkValidation() {
            var password = document.getElementById("password").value;
            var lowercase = /[a-z]/.test(password);
            var uppercase = /[A-Z]/.test(password);
            var digit = /\d/.test(password);
            var specialChar = /[!@#$%^&*(),.?":{}|<>]/.test(password);
            var minLength = password.length >= 8;

            document.getElementById("lowercase").checked = lowercase;
            document.getElementById("uppercase").checked = uppercase;
            document.getElementById("digit").checked = digit;
            document.getElementById("specialChar").checked = specialChar;
            document.getElementById("minLength").checked = minLength;
        }
    </script>
</head>
